version: '3.8'

services:
  # Main application
  autogen-app:
    build: .
    container_name: autogen-enterprise
    ports:
      - "8080:8080"  # API port
      - "8000:8000"  # Metrics port
    environment:
      - ENV_FILE=/app/.env
    volumes:
      - ./.env:/app/.env:ro
      - ./generated_code:/app/generated_code
      - ./logs:/app/logs
      - /var/run/docker.sock:/var/run/docker.sock:ro  # For Docker-in-Docker
    depends_on:
      - redis
      - ollama
    networks:
      - autogen-network
    restart: unless-stopped

  # Redis for caching and state management
  redis:
    image: redis:7-alpine
    container_name: autogen-redis
    ports:
      - "6379:6379"
    volumes:
      - redis-data:/data
    command: redis-server --appendonly yes
    networks:
      - autogen-network
    restart: unless-stopped

  # Ollama for local LLM models
  ollama:
    image: ollama/ollama:latest
    container_name: autogen-ollama
    ports:
      - "11434:11434"
    volumes:
      - ollama-data:/root/.ollama
    networks:
      - autogen-network
    restart: unless-stopped
    deploy:
      resources:
        limits:
          memory: 16G
        reservations:
          devices:
            - driver: nvidia
              count: all
              capabilities: [gpu]  # Optional: for GPU support

  # Prometheus for metrics
  prometheus:
    image: prom/prometheus:latest
    container_name: autogen-prometheus
    ports:
      - "9090:9090"
    volumes:
      - ./config/prometheus.yml:/etc/prometheus/prometheus.yml:ro
      - prometheus-data:/prometheus
    command:
      - '--config.file=/etc/prometheus/prometheus.yml'
      - '--storage.tsdb.path=/prometheus'
      - '--web.console.libraries=/usr/share/prometheus/console_libraries'
      - '--web.console.templates=/usr/share/prometheus/consoles'
    networks:
      - autogen-network
    restart: unless-stopped

  # Grafana for visualization
  grafana:
    image: grafana/grafana:latest
    container_name: autogen-grafana
    ports:
      - "3000:3000"
    environment:
      - GF_SECURITY_ADMIN_PASSWORD=admin
      - GF_USERS_ALLOW_SIGN_UP=false
    volumes:
      - grafana-data:/var/lib/grafana
      - ./config/grafana/dashboards:/etc/grafana/provisioning/dashboards:ro
      - ./config/grafana/datasources:/etc/grafana/provisioning/datasources:ro
    depends_on:
      - prometheus
    networks:
      - autogen-network
    restart: unless-stopped

networks:
  autogen-network:
    driver: bridge

volumes:
  redis-data:
  ollama-data:
  prometheus-data:
  grafana-data:
